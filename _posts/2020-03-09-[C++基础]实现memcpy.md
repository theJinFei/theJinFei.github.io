## 内存拷贝，注意覆盖情况 memcopy

```C++
void* memcpy(void* des, const void* src, int length){
    assert(des != NULL &&  src != NULL);
    char* d;
    const char* s;
    if(src + length < des || des < src){
        d = des;
        s = src;
        while(length > 0){
            *d = *s;
            d++;
            s++;
            length--;
        }
    }else{
        d = (char*)(des + length - 1);
        s = (char*)(src + length - 1);
        while(length > 0){
            *d = *s;
            d--;
            s--;
            length--;
        }
    }
    return des;
}
```
